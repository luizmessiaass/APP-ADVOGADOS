**2. NotificationsRepositoryImpl.kt:**
```kotlin
class NotificationsRepositoryImpl @Inject constructor(
    private val remoteDataSource: NotificationsRemoteDataSource
) : NotificationsRepository {

    override suspend fun getNotifications(): Result<Pair<List<NotificationItem>, List<NotificationItem>>> {
        return try {
            val all = remoteDataSource.getNotifications(unreadOnly = false, limit = 50)
            val thirtyDaysAgo = Instant.now().minus(30, ChronoUnit.DAYS)
            val filtered = all.filter { Instant.parse(it.createdAt).isAfter(thirtyDaysAgo) }
            val unread = filtered.filter { !it.isRead }
            val read = filtered.filter { it.isRead }
            Result.success(Pair(unread, read))
        } catch (e: Exception) {
            Result.failure(e)
        }
    }

    override suspend fun markAllRead() = try {
        remoteDataSource.markAllRead()
        Result.success(Unit)
    } catch (e: Exception) { Result.failure(e) }

    override suspend fun markRead(id: String) = try {
        remoteDataSource.markRead(id)
        Result.success(Unit)
    } catch (e: Exception) { Result.failure(e) }
}
```

**3. NotificationsViewModel.kt:**
```kotlin
@HiltViewModel
class NotificationsViewModel @Inject constructor(
    private val repository: NotificationsRepository
) : ViewModel() {

    private val _uiState = MutableStateFlow<NotificationsUiState>(NotificationsUiState.Loading)
    val uiState: StateFlow<NotificationsUiState> = _uiState.asStateFlow()

    val unreadCount: StateFlow<Int> = uiState
        .map { state -> if (state is NotificationsUiState.Success) state.unread.size else 0 }
        .stateIn(viewModelScope, SharingStarted.Eagerly, 0)

    init { loadNotifications() }

    fun loadNotifications() {
        viewModelScope.launch {
            _uiState.value = NotificationsUiState.Loading
            repository.getNotifications()
                .onSuccess { (unread, read) ->
                    _uiState.value = NotificationsUiState.Success(unread, read)
                }
                .onFailure {
                    _uiState.value = NotificationsUiState.Error("Não foi possível carregar as notificações")
                }
        }
    }

    fun markAllRead() {
        viewModelScope.launch {
            repository.markAllRead()
            loadNotifications()  // atualiza estado para zerar badge
        }
    }

    fun markRead(id: String) {
        viewModelScope.launch {
            repository.markRead(id)
            loadNotifications()
        }
    }
}
```

@Composable
private fun NotificationItemCard(
    item: NotificationItem,
    onTap: (processoId: String?) -> Unit
) {
    Card(
        onClick = { onTap(item.processoId) },
        modifier = Modifier.fillMaxWidth().padding(horizontal = 16.dp, vertical = 4.dp)

        Row(modifier = Modifier.padding(16.dp)) {

                Text(

                    color = MaterialTheme.colorScheme.outline
                )
            }
        }
    }
}

fun formatRelativeTime(createdAt: String): String {
    val instant = Instant.parse(createdAt)
    val now = Instant.now()
    val minutes = Duration.between(instant, now).toMinutes()
    return when {
        minutes < 60 -> "há $minutes minuto${if (minutes != 1L) "s" else ""}"
        minutes < 1440 -> "há ${minutes / 60} hora${if (minutes / 60 != 1L) "s" else ""}"
        else -> "há ${minutes / 1440} dia${if (minutes / 1440 != 1L) "s" else ""}"
    }
}
