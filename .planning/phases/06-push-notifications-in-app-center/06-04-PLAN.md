```kotlin
@AndroidEntryPoint
class PortalMessagingService : FirebaseMessagingService() {

    @Inject lateinit var deviceTokenApi: DeviceTokenApi
    @Inject lateinit var dataStore: DataStore<Preferences>

    override fun onNewToken(token: String) {
        // Persistir token localmente
        CoroutineScope(Dispatchers.IO).launch {
            dataStore.edit { it[FCM_TOKEN_REGISTERED] = token }
            // Registrar no backend — gerar deviceId estável
            val deviceId = Settings.Secure.getString(contentResolver, Settings.Secure.ANDROID_ID)
            deviceTokenApi.registerToken(RegisterTokenRequest(token = token, deviceId = deviceId))
        }
    }

    override fun onMessageReceived(remoteMessage: RemoteMessage) {
        val notificationId = remoteMessage.data["notification_id"] ?: return
        val processoId = remoteMessage.data["processo_id"] ?: return
        val titulo = remoteMessage.data["titulo"] ?: "Nova movimentação"
        val corpo = remoteMessage.data["corpo"] ?: ""

        CoroutineScope(Dispatchers.IO).launch {
            val shownIds = dataStore.data.first()[SHOWN_NOTIFICATION_IDS] ?: emptySet()
            if (notificationId in shownIds) return@launch  // deduplicação

            dataStore.edit { prefs ->
                val current = prefs[SHOWN_NOTIFICATION_IDS] ?: emptySet()
                // Manter apenas últimas 100 IDs para evitar crescimento ilimitado
                val updated = if (current.size >= 100) current.drop(50).toSet() + notificationId
                              else current + notificationId
                prefs[SHOWN_NOTIFICATION_IDS] = updated
            }

            showLocalNotification(notificationId.hashCode(), titulo, corpo, processoId)
        }
    }

    private fun showLocalNotification(id: Int, titulo: String, corpo: String, processoId: String) {
        val intent = Intent(Intent.ACTION_VIEW, Uri.parse("portaljuridico://processo/$processoId"))
        val pendingIntent = PendingIntent.getActivity(
            this, id, intent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
        )
        val notification = NotificationCompat.Builder(this, ClienteApplication.CHANNEL_ID)
            .setSmallIcon(R.drawable.ic_notification)  // usar ícone existente do projeto
            .setContentTitle(titulo)
            .setContentText(corpo)
            .setContentIntent(pendingIntent)
            .setAutoCancel(true)
            .build()
        NotificationManagerCompat.from(this).notify(id, notification)
    }
}
```

**2. NotificationPollWorker.kt:**
```kotlin
@HiltWorker
class NotificationPollWorker @AssistedInject constructor(
    @Assisted context: Context,
    @Assisted workerParams: WorkerParameters,
    private val remoteDataSource: NotificationsRemoteDataSource,
    private val dataStore: DataStore<Preferences>
) : CoroutineWorker(context, workerParams) {

    override suspend fun doWork(): Result {
        return try {
            val unread = remoteDataSource.getUnreadNotifications()
            val shownIds = dataStore.data.first()[SHOWN_NOTIFICATION_IDS] ?: emptySet()
            val newOnes = unread.filter { it.id !in shownIds }

            if (newOnes.isNotEmpty()) {
                showBatchLocalNotification(newOnes.size)
                dataStore.edit { prefs ->
                    val current = prefs[SHOWN_NOTIFICATION_IDS] ?: emptySet()
                    prefs[SHOWN_NOTIFICATION_IDS] = (current + newOnes.map { it.id }.toSet())
                        .takeLast(100).toSet()
                }
            }
            Result.success()
        } catch (e: IOException) {
            Result.retry()
        } catch (e: Exception) {
            Result.failure()
        }
    }

    private fun showBatchLocalNotification(count: Int) {
        val intent = Intent(Intent.ACTION_VIEW, Uri.parse("portaljuridico://notificacoes"))
        val pendingIntent = PendingIntent.getActivity(
            applicationContext, POLL_NOTIFICATION_ID, intent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
        )
        val notification = NotificationCompat.Builder(applicationContext, ClienteApplication.CHANNEL_ID)
            .setSmallIcon(R.drawable.ic_notification)
            .setContentTitle("Portal Jurídico")
            .setContentText("Você tem $count nova(s) movimentação(ões)")
            .setContentIntent(pendingIntent)
            .setAutoCancel(true)
            .build()
        NotificationManagerCompat.from(applicationContext).notify(POLL_NOTIFICATION_ID, notification)
    }

    companion object {
        const val POLL_NOTIFICATION_ID = 1001

        fun enqueue(context: Context) {
            WorkManager.getInstance(context).enqueueUniquePeriodicWork(
                "notification_poll",
                ExistingPeriodicWorkPolicy.KEEP,
                PeriodicWorkRequestBuilder<NotificationPollWorker>(15, TimeUnit.MINUTES)
                    .setConstraints(
                        Constraints.Builder()
                            .setRequiredNetworkType(NetworkType.CONNECTED)
                            .build()
                    )
                    .build()
            )
        }
    }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| FCM payload data → onMessageReceived | Dados vêm da FCM (backend confiável) mas devem ser validados antes de usar em PendingIntent |
| deep_link em payload FCM → PendingIntent | Se deep_link vier do payload, pode ser manipulado — usar apenas processoId para construir o link |
| android.permission.REQUEST_IGNORE_BATTERY_OPTIMIZATIONS | Declaração obrigatória; manifesto público |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-13 | Tampering | deep_link do payload FCM usado diretamente no PendingIntent | mitigate | Nunca usar `remoteMessage.data["deep_link"]` diretamente — construir o URI com `processoId` extraído e validado como UUID antes de montar `portaljuridico://processo/$processoId` |
| T-06-14 | Elevation of Privilege | PendingIntent.FLAG_MUTABLE em notificações | mitigate | Usar sempre `PendingIntent.FLAG_IMMUTABLE` (obrigatório para Android 12+, targetSdk 36) |
| T-06-15 | Information Disclosure | FCM token em DataStore legível por outros apps | accept | DataStore usa armazenamento privado do app (MODE_PRIVATE); sem acesso cross-app sem root |
| T-06-16 | Denial of Service | SHOWN_NOTIFICATION_IDS crescendo ilimitado no DataStore | mitigate | Limitar a 100 IDs (descartando os mais antigos); implementado no código via `.takeLast(100)` |
</threat_model>
