```sql
-- Migration 0008: Add cliente_usuario_id to processos for per-client process ownership
-- Required by Phase 5 (APP-02): clients must only see their own processes

-- Add the column (nullable for backwards compat with existing processos rows)
ALTER TABLE processos
  ADD COLUMN IF NOT EXISTS cliente_usuario_id UUID REFERENCES usuarios(id) ON DELETE SET NULL;

-- Index for efficient per-client queries
CREATE INDEX IF NOT EXISTS idx_processos_cliente_usuario_id
  ON processos (cliente_usuario_id);

-- Add escritorio WhatsApp phone field if not present
ALTER TABLE escritorios
  ADD COLUMN IF NOT EXISTS telefone_whatsapp TEXT;

-- Drop and recreate the RLS SELECT policy for role='cliente'
-- Old policy (Phase 1): only checks tenant_id
-- New policy: checks tenant_id AND cliente_usuario_id = requesting user
DROP POLICY IF EXISTS "clientes_select_own_processos" ON processos;

CREATE POLICY "clientes_select_own_processos"
  ON processos
  FOR SELECT
  TO authenticated
  USING (
    (auth.jwt() -> 'app_metadata' ->> 'role') = 'cliente'
    AND tenant_id = (auth.jwt() -> 'app_metadata' ->> 'tenant_id')::uuid
    AND (cliente_usuario_id = auth.uid())
  );
```

In `apps/api/src/routes/processos.ts`, add a GET list handler before the existing GET :id handler:

```typescript
// GET /api/v1/processos — list all processes for the authenticated user
fastify.get('/', {
  preHandler: [fastify.authenticate],
  schema: {
    response: {
      200: z.object({
        success: z.literal(true),
        data: z.array(z.object({
          id: z.string().uuid(),
          numero_cnj: z.string(),
          tribunal: z.string().nullable(),
          ultima_sincronizacao: z.string().nullable(),
          desatualizado: z.boolean(),
          status: z.string().nullable(),  // latest movimentacao.status
        }))
      })
    }
  }
}, async (request, reply) => {
  // Use the user's JWT-authenticated Supabase client so RLS applies
  const { data, error } = await fastify.supabase
    .from('processos')
    .select(`
      id, numero_cnj, tribunal, ultima_sincronizacao, desatualizado,
      movimentacoes(status, data_hora)
    `)
    .order('created_at', { ascending: false });

  if (error) throw error;

  const mapped = data.map(p => ({
    id: p.id,
    numero_cnj: p.numero_cnj,
    tribunal: p.tribunal,
    ultima_sincronizacao: p.ultima_sincronizacao,
    desatualizado: p.desatualizado,
    // Latest movimentacao status (most recent by data_hora)
    status: p.movimentacoes
      ?.sort((a: any, b: any) => new Date(b.data_hora).getTime() - new Date(a.data_hora).getTime())
      ?.[0]?.status ?? null
  }));

  return reply.status(200).send({ success: true, data: mapped });
});
```

Create new file `apps/api/src/routes/processos/movimentacoes.ts`:

```typescript
import { FastifyInstance } from 'fastify';
import { z } from 'zod';

export async function movimentacoesRoutes(fastify: FastifyInstance) {
  // GET /api/v1/processos/:id/movimentacoes
  fastify.get('/:id/movimentacoes', {
    preHandler: [fastify.authenticate],
    schema: {
      params: z.object({ id: z.string().uuid() }),
      response: {
        200: z.object({
          success: z.literal(true),
          data: z.array(z.object({
            id: z.string().uuid(),
            data_hora: z.string(),
            descricao_original: z.string(),
            status: z.string().nullable(),
            explicacao: z.string().nullable(),
            proxima_data: z.string().nullable(),
            impacto: z.string().nullable(),
          }))
        })
      }
    }
  }, async (request, reply) => {
    const { id } = request.params as { id: string };

    // Verify process access (RLS will block if not allowed — returns empty array)
    const { data: processo } = await fastify.supabase
      .from('processos')
      .select('id')
      .eq('id', id)
      .single();

    if (!processo) {
      return reply.status(404).send({ success: false, error: 'Processo não encontrado' });
    }

    const { data, error } = await fastify.supabase
      .from('movimentacoes')
      .select('id, data_hora, descricao_original, status, explicacao, proxima_data, impacto')
      .eq('processo_id', id)
      .order('data_hora', { ascending: false });

    if (error) throw error;

    return reply.status(200).send({ success: true, data: data ?? [] });
  });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| JWT → processos RLS | Authenticated user's JWT drives row-level filtering; tampered JWT claims would expose wrong tenant's data |
| WhatsApp phone in API response | telefone_whatsapp sourced from backend (admin-controlled), not user input — no injection risk at this layer |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01-01 | Spoofing | GET /api/v1/processos — cliente sees all tenant processos | mitigate | RLS policy requires `cliente_usuario_id = auth.uid()` (Task 1). Without this, any authenticated user in the tenant can see all processes. |
| T-05-01-02 | Tampering | POST /processos — escritório assigns process to wrong cliente_usuario_id | accept | Only admin_escritorio/advogado role can POST; they are trusted to assign correctly. No cross-tenant risk since RLS still enforces tenant_id. |
| T-05-01-03 | Information Disclosure | GET /processos/:id/movimentacoes — cliente accesses another client's movimentacoes | mitigate | Route fetches processo first; RLS on processos blocks access if cliente_usuario_id doesn't match. Returns 404 if RLS blocks. |
| T-05-01-04 | Denial of Service | Unbounded SELECT on movimentacoes for large processes | accept | Phase 5 MVP; add pagination in Phase 6 if needed. Risk: slow queries on processes with 1000+ movimentacoes. Acceptable for launch. |
</threat_model>
