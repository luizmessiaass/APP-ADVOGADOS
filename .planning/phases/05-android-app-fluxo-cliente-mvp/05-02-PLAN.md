    ```kotlin
    @Composable
    fun ClienteNavGraph(navController: NavHostController, startDestination: String) {
        NavHost(navController = navController, startDestination = startDestination) {
            composable(Routes.LOGIN) { LoginScreen(navController = navController) }
            composable(Routes.ONBOARDING) { OnboardingScreen(navController = navController) }
            composable(Routes.LGPD_CONSENT) { LgpdConsentScreen(navController = navController) }
            composable(Routes.PROCESSO_LIST) { ProcessoListScreen(navController = navController) }
            composable(Routes.PROCESSO_DETAIL_WITH_ID) { backStackEntry ->
                val processoId = backStackEntry.arguments?.getString("processoId") ?: return@composable
                ProcessoDetailScreen(processoId = processoId, navController = navController)
            }
        }
    }
    ```

12. `SplashViewModel.kt`:
    ```kotlin
    @HiltViewModel
    class SplashViewModel @Inject constructor(
        @ApplicationContext private val context: Context,
    ) : ViewModel() {
        private val dataStore: DataStore<Preferences> = context.clienteDataStore

        val startDestination: StateFlow<String?> = dataStore.data
            .map { prefs ->
                val token = prefs[TOKEN_KEY]
                val onboardingSeen = prefs[ONBOARDING_SEEN_KEY] ?: false
                val lgpdAccepted = prefs[LGPD_ACCEPTED_KEY] ?: false
                when {
                    token == null -> Routes.LOGIN
                    !onboardingSeen -> Routes.ONBOARDING
                    !lgpdAccepted -> Routes.LGPD_CONSENT
                    else -> Routes.PROCESSO_LIST
                }
            }
            .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), initialValue = null)
    ```
