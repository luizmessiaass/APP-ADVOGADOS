```typescript
/**
 * Rotas de processos jurídicos.
 * (DATAJUD-01, DATAJUD-02, DATAJUD-09, D-07/D-08/D-09, D-19/D-21)
 *
 * Endpoints:
 * GET  /api/v1/processos/:id        — dados do processo com staleness flag
 * POST /api/v1/processos            — cadastrar processo (valida CNJ, agenda sync)
 * POST /api/v1/processos/:id/sync   — disparar sync manual
 */
import type { FastifyPluginAsync } from 'fastify';
import { assertCNJValido, CNJInvalidoError, extrairCodigoTribunal } from '../datajud/cnj-validator.js';
import { resolverTribunal } from '../datajud/tribunal-map.js';
import { calcularTier, agendarSyncProcesso } from '../workers/scheduler.js';

// Staleness threshold: 72 horas em ms (D-07)
const STALENESS_THRESHOLD_MS = 72 * 60 * 60 * 1000;

function calcularDesatualizado(ultimaSincronizacao: string | null): boolean {
  if (!ultimaSincronizacao) return false; // Nunca sincronizou — badge diferente, não "desatualizado"
  return Date.now() - new Date(ultimaSincronizacao).getTime() > STALENESS_THRESHOLD_MS;
}

export const processosRoutes: FastifyPluginAsync = async (fastify) => {
  // GET /api/v1/processos/:id — buscar processo por ID (dados em cache)
  fastify.get<{ Params: { id: string } }>('/processos/:id', async (request, reply) => {
    const { id } = request.params;
    const tenantId = request.tenantId; // injetado pelo middleware de tenant (Phase 1)

    const { data, error } = await request.supabase // cliente autenticado (respeitando RLS)
      .from('processos')
      .select(`
        id,
        numero_cnj,
        tribunal,
        tier_refresh,
        ultima_sincronizacao,
        sincronizado,
        created_at,
        updated_at
      `)
      .eq('id', id)
      .eq('tenant_id', tenantId)
      .single();

    if (error || !data) {
      return reply.status(404).send({
        success: false,
        error: 'Processo não encontrado',
        code: 'PROCESSO_NOT_FOUND',
      });
    }

    const desatualizado = calcularDesatualizado(data.ultima_sincronizacao);

    return reply.status(200).send({
      success: true,
      data: {
        id: data.id,
        numero_cnj: data.numero_cnj,
        tribunal: data.tribunal,
        tier_refresh: data.tier_refresh,
        ultima_sincronizacao: data.ultima_sincronizacao,
        sincronizado: data.sincronizado,
        desatualizado,                     // D-08: flag para UI mostrar badge (DATAJUD-09)
        created_at: data.created_at,
        updated_at: data.updated_at,
      },
    });
  });

  // POST /api/v1/processos — cadastrar novo processo
  fastify.post<{
    Body: { numero_cnj: string; }
  }>('/processos', {
    schema: {
      body: {
        type: 'object',
        required: ['numero_cnj'],
        properties: {
          numero_cnj: { type: 'string' },
        },
      },
    },
  }, async (request, reply) => {
    const { numero_cnj } = request.body;
    const tenantId = request.tenantId;

    // DATAJUD-01: validar CNJ antes de qualquer I/O
    try {
      assertCNJValido(numero_cnj);
    } catch (err) {
      if (err instanceof CNJInvalidoError) {
        return reply.status(422).send({
          success: false,
          error: err.message,
          code: err.code, // 'INVALID_CNJ'
        });
      }
      throw err;
    }

    // Resolver tribunal a partir do CNJ
    let tribunal: string;
    try {
      const codigoJT = extrairCodigoTribunal(numero_cnj);
      tribunal = resolverTribunal(codigoJT);
    } catch {
      return reply.status(422).send({
        success: false,
        error: `Tribunal não suportado para o número CNJ informado`,
        code: 'TRIBUNAL_NOT_SUPPORTED',
      });
    }

    // Inserir processo (RLS garante que tenant_id == request.tenantId)
    const { data, error } = await request.supabase
      .from('processos')
      .insert({
        tenant_id: tenantId,
        numero_cnj,
        tribunal,
        tier_refresh: 'cold', // começa como cold — sem movimentações conhecidas
        sincronizado: false,
      })
      .select('id, numero_cnj, tribunal, tier_refresh, created_at')
      .single();

    if (error) {
      if (error.code === '23505') { // unique violation
        return reply.status(409).send({
          success: false,
          error: 'Processo já cadastrado para este escritório',
          code: 'PROCESSO_ALREADY_EXISTS',
        });
      }
      request.log.error({ error, tenantId }, 'Erro ao criar processo');
      return reply.status(500).send({
        success: false,
        error: 'Erro interno ao cadastrar processo',
        code: 'INTERNAL_ERROR',
      });
    }

    // Agendar primeiro sync (tier 'cold' — processo recém-cadastrado)
    const queue = request.datajudQueue; // injetado via fastify.decorate ou plugin
    if (queue) {
      await agendarSyncProcesso(queue, data.id, numero_cnj, tenantId, 'cold');
    }

    return reply.status(201).send({ success: true, data });
  });

  // POST /api/v1/processos/:id/sync — disparar sync manual
  fastify.post<{ Params: { id: string } }>('/processos/:id/sync', async (request, reply) => {
    const { id } = request.params;
    const tenantId = request.tenantId;

    // Verificar que o processo existe e pertence ao tenant
    const { data, error } = await request.supabase
      .from('processos')
      .select('id, numero_cnj, tier_refresh')
      .eq('id', id)
      .eq('tenant_id', tenantId)
      .single();

    if (error || !data) {
      return reply.status(404).send({
        success: false,
        error: 'Processo não encontrado',
        code: 'PROCESSO_NOT_FOUND',
      });
    }

    const queue = request.datajudQueue;
    if (queue) {
      await agendarSyncProcesso(
        queue,
        data.id,
        data.numero_cnj,
        tenantId,
        data.tier_refresh as 'hot' | 'warm' | 'cold'
      );
    }

    return reply.status(202).send({
      success: true,
      message: 'Sincronização agendada',
    });
  });
};
```

import { createBullBoard } from '@bull-board/api';
import { BullMQAdapter } from '@bull-board/api/bullMQAdapter.js';
import { FastifyAdapter } from '@bull-board/fastify';
import { getDatajudQueue } from './queues/datajud-queue.js';

// Guard de autenticação simples (D-11 — Bearer token de env var)
fastify.addHook('onRequest', async (request, reply) => {
  if (!request.url.startsWith('/admin/queues')) return;

  const authHeader = request.headers.authorization ?? '';
  const token = authHeader.startsWith('Bearer ') ? authHeader.slice(7) : '';
  const adminToken = process.env.ADMIN_TOKEN;

  if (!adminToken || token !== adminToken) {
    return reply.status(401).send({
      success: false,
      error: 'Unauthorized',
      code: 'UNAUTHORIZED',
    });
  }
});

// Registrar Bull Board (D-12 — roteado pelo serviço API)
const bullBoardAdapter = new FastifyAdapter();
bullBoardAdapter.setBasePath('/admin/queues');

createBullBoard({
  queues: [new BullMQAdapter(getDatajudQueue(redis))],
  serverAdapter: bullBoardAdapter,
});

await fastify.register(bullBoardAdapter.registerPlugin(), {
  prefix: '/admin/queues',
  basePath: '/admin/queues',
});

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| App Android (JWT) → GET /processos/:id | Input não-confiável: id pode ser UUID de outro tenant — RLS filtra |
| Usuário (JWT) → POST /processos (numero_cnj) | Input não-confiável: número CNJ pode ser inválido ou de outro processo |
| Admin browser → /admin/queues | ADMIN_TOKEN deve ser forte e mantido em segredo |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-17 | Information Disclosure | GET /processos/:id | mitigate | RLS via Supabase client autenticado (não supabaseAdmin) — tenant só vê seus processos; .eq('tenant_id', tenantId) como segunda camada |
| T-02-18 | Tampering | POST /processos (numero_cnj) | mitigate | assertCNJValido chamado antes de qualquer insert; resposta 422 com code INVALID_CNJ retornada sem I/O externo |
| T-02-19 | Elevation of Privilege | /admin/queues | mitigate | Bearer token de 32 bytes (openssl rand -hex 32) em env var; sem exposição no painel de escritório; rate limiting herdado do Fastify Phase 1 |
| T-02-20 | Information Disclosure | /admin/queues (Bull Board) | accept | Bull Board expõe job data (pode conter numeroCNJ e tenantId); aceitável pois o endpoint é admin-only com token forte; não exposto externamente via URL pública sem token |
| T-02-21 | Denial of Service | POST /processos/:id/sync | mitigate | upsertJobScheduler é idempotente — múltiplas chamadas ao sync manual não criam storm de jobs; rate limiting do Fastify aplica por IP |
</threat_model>
