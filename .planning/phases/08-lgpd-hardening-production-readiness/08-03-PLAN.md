    ```typescript
    import type { FastifyInstance } from 'fastify'
    import { Type } from '@sinclair/typebox'
    import { supabaseAdmin } from '../../lib/supabase.js'
    import { env } from '../../config.js'

    const WebhookBody = Type.Object({
      event: Type.String(),
      tenant_id: Type.String(),
      event_id: Type.String(),
    }, { additionalProperties: true })

    export async function billingWebhookRoutes(app: FastifyInstance): Promise<void> {
      app.post('/', {
        schema: { body: WebhookBody },
      }, async (req, reply) => {
        // Auth check
        const secret = req.headers['x-webhook-secret']
        if (!secret || secret !== env.WEBHOOK_SECRET) {
          return reply.code(401).send({ success: false, error: 'Unauthorized', code: 'INVALID_SECRET' })
        }

        const { event, tenant_id, event_id } = req.body as {
          event: string; tenant_id: string; event_id: string
        }

        // Idempotency: check for existing event_id (UNIQUE constraint handles race conditions too)
        const { data: existing } = await supabaseAdmin
          .from('billing_events')
          .select('id')
          .eq('event_id', event_id)
          .single()

        if (existing) {
          // Already processed — idempotent 200
          return reply.code(200).send({ success: true, idempotent: true })
        }

        // Insert new event record
        const { error: insertError } = await supabaseAdmin
          .from('billing_events')
          .insert({ event_id, tenant_id, event_type: event, payload: req.body })

        if (insertError) {
          // Handle unique constraint violation (race condition) — still idempotent
          if (insertError.code === '23505') {
            return reply.code(200).send({ success: true, idempotent: true })
          }
          return reply.code(500).send({ success: false, error: 'Internal error', code: 'INSERT_ERROR' })
        }

        // TODO Phase 7: process billing event (update tenant subscription_status etc.)

        return reply.code(200).send({ success: true })
      })
    }
    ```

    import { describe, it, expect, vi, beforeAll, afterAll } from 'vitest'
    import Fastify from 'fastify'
    import { billingWebhookRoutes } from './billing.js'

    // Mock supabaseAdmin for billing_events queries
    const mockSingle = vi.fn()
    const mockInsert = vi.fn()

    vi.mock('../../lib/supabase.js', () => ({
      supabaseAdmin: {
        from: vi.fn().mockReturnValue({
          select: vi.fn().mockReturnThis(),
          eq: vi.fn().mockReturnThis(),
          single: mockSingle,
          insert: mockInsert,
        }),
      },
    }))

    vi.mock('../../config.js', () => ({
      env: {
        WEBHOOK_SECRET: 'test-secret-123',
        TERMS_VERSION: '2026-04-16',
      },
    }))

    describe('POST /api/webhooks/billing (BILLING-03, BILLING-04)', () => {
      let app: ReturnType<typeof Fastify>

      beforeAll(async () => {
        app = Fastify({ logger: false })
        await app.register(billingWebhookRoutes)
        await app.ready()
      })

      afterAll(() => app.close())

      beforeEach(() => {
        vi.clearAllMocks()
      })

      it('missing X-Webhook-Secret header — retorna 401', async () => {
        const res = await app.inject({
          method: 'POST',
          url: '/',
          payload: { event: 'payment.failed', tenant_id: 'tenant-1', event_id: 'evt-001' },
        })
        expect(res.statusCode).toBe(401)
        expect(JSON.parse(res.body).code).toBe('INVALID_SECRET')
      })

      it('X-Webhook-Secret errado — retorna 401', async () => {
        const res = await app.inject({
          method: 'POST',
          url: '/',
          payload: { event: 'payment.failed', tenant_id: 'tenant-1', event_id: 'evt-002' },
          headers: { 'x-webhook-secret': 'wrong-secret' },
        })
        expect(res.statusCode).toBe(401)
      })

      it('evento payment.failed valido — gravado em billing_events, retorna 200', async () => {
        // event_id not seen before: single returns null, insert succeeds
        mockSingle.mockResolvedValueOnce({ data: null, error: { code: 'PGRST116' } })
        mockInsert.mockResolvedValueOnce({ error: null })

        const res = await app.inject({
          method: 'POST',
          url: '/',
          payload: { event: 'payment.failed', tenant_id: 'tenant-1', event_id: 'evt-003' },
          headers: { 'x-webhook-secret': 'test-secret-123' },
        })
        expect(res.statusCode).toBe(200)
        expect(JSON.parse(res.body).success).toBe(true)
        expect(mockInsert).toHaveBeenCalled()
      })

      it('replay de event_id duplicado — idempotente, retorna 200 sem reprocessar', async () => {
        // Same event_id replayed: single returns existing row (already processed)
        // vi.clearAllMocks() in beforeEach resets all call counts — this test is self-contained
        mockSingle.mockResolvedValueOnce({ data: { id: 'existing-row-id' }, error: null })

        // Capture insert call count BEFORE making the request
        const insertCallsBefore = mockInsert.mock.calls.length

        const res = await app.inject({
          method: 'POST',
          url: '/',
          payload: { event: 'payment.failed', tenant_id: 'tenant-1', event_id: 'evt-003' },
          headers: { 'x-webhook-secret': 'test-secret-123' },
        })
        expect(res.statusCode).toBe(200)
        expect(JSON.parse(res.body).idempotent).toBe(true)
        // insert should NOT be called on replay — call count must equal count captured before the request
        expect(mockInsert.mock.calls.length).toBe(insertCallsBefore)
      })
    })
