```typescript
import fp from 'fastify-plugin'
import type { FastifyPluginAsync } from 'fastify'
import type { Redis } from 'ioredis'
import { supabaseAdmin } from '../lib/supabase.js'

const CACHE_TTL_SECONDS = 30

export const ENTITLEMENT_SKIP_ROUTES = new Set([
  '/health',
  '/api/v1/auth/login',
  '/api/v1/auth/signup',
  '/api/webhooks/billing',
  '/api/v1/tenant/status',
])

export async function invalidateTenantStatusCache(redis: Redis, tenantId: string): Promise<void> {
  await redis.del(`tenant:status:${tenantId}`)
}

const entitlementPlugin: FastifyPluginAsync<{ redis: Redis }> = async (fastify, opts) => {
  fastify.addHook('preHandler', async (request, reply) => {
    const url = request.url.split('?')[0]
    if (ENTITLEMENT_SKIP_ROUTES.has(url)) return
    if (!request.user?.tenant_id) return

    const tenantId = request.user.tenant_id
    const cacheKey = `tenant:status:${tenantId}`

    let status: string
    const cached = await opts.redis.get(cacheKey)

    if (cached) {
      status = cached
    } else {
      const { data, error } = await supabaseAdmin
        .from('escritorios')
        .select('status')
        .eq('id', tenantId)
        .single()

      if (error || !data) {
        // Fail closed: unknown tenant treated as suspended
        status = 'suspended'
      } else {
        status = data.status
      }
      await opts.redis.setex(cacheKey, CACHE_TTL_SECONDS, status)
    }

    if (status === 'suspended') {
      return reply.code(402).send({
        error: 'subscription_required',
        tenant_status: 'suspended',
        message: 'Assinatura suspensa. Entre em contato com o Portal Juridico.',
      })
    }

    if (status === 'read_only') {
      const method = request.method.toUpperCase()
      if (['POST', 'PUT', 'PATCH', 'DELETE'].includes(method)) {
        return reply.code(402).send({
          error: 'subscription_required',
          tenant_status: 'read_only',
          message: 'Conta em modo leitura. Regularize a assinatura para retomar operacoes de escrita.',
        })
      }
    }
    // pending, trial, active, grace — pass through
  })
}

export default fp(entitlementPlugin, { name: 'entitlement', dependencies: ['auth'] })
```

not open. This prevents a DB error from giving suspended tenants access.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Redis cache → entitlement decision | Stale cache could allow a suspended tenant brief access (max 30s TTL) |
| DB unavailable → entitlement decision | Network error fetching status from Supabase |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-7-04 | Tampering | Redis cache stale status | mitigate | 30s TTL + explicit invalidateTenantStatusCache() call on every admin status change — documented in code |
| T-7-05 | Denial of Service | entitlement Redis query | mitigate | Fail closed (treat as suspended) on Redis error — tenant cannot bypass entitlement via Redis outage |
| T-7-06 | Elevation of Privilege | super_admin in TenantUser | mitigate | super_admin added to role union but does NOT bypass entitlement — super_admin's own tenant status is still checked. Admin endpoints explicitly require super_admin role check in addition to entitlement |
| T-7-07 | Information Disclosure | 402 response body | accept | Response body reveals tenant_status — acceptable because the authenticated user is entitled to know their own billing state |
</threat_model>
