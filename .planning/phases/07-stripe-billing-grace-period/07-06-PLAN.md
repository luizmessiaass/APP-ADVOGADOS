```typescript
import type { FastifyInstance } from 'fastify'
import { supabaseAdmin } from '../../lib/supabase.js'

export async function tenantStatusRoutes(app: FastifyInstance): Promise<void> {
  app.get('/status', async (request, reply) => {
    const tenantId = request.user.tenant_id

    const { data, error } = await supabaseAdmin
      .from('escritorios')
      .select('status, grace_banner, grace_period_started_at')
      .eq('id', tenantId)
      .single()

    if (error || !data) {
      return reply.code(500).send({ success: false, error: 'Tenant nao encontrado', code: 'TENANT_NOT_FOUND' })
    }

    let daysUntilSuspension: number | null = null
    if (data.grace_period_started_at) {
      const started = new Date(data.grace_period_started_at).getTime()
      const daysSinceStart = Math.floor((Date.now() - started) / 86_400_000)
      daysUntilSuspension = Math.max(0, 14 - daysSinceStart)
    }

    return reply.send({
      status: data.status,
      grace_banner: data.grace_banner,
      grace_period_started_at: data.grace_period_started_at,
      days_until_suspension: daysUntilSuspension,
    })
  })
}
```

| T-7-20 | Elevation of Privilege | Suspended app_escritorio user accesses write APIs | mitigate | Entitlement middleware (Plan 02) blocks write endpoints at the server; Android UI disabling is defense-in-depth only |
| T-7-21 | Tampering | GET /api/v1/tenant/status bypassed by modified APK | accept | Server-side entitlement middleware is the authoritative enforcement layer; client-side UI is UX-only |
| T-7-22 | Availability | WorkManager not firing on Xiaomi/Samsung | mitigate | Same battery optimization guidance as Phase 6 (onboarding tela 5); NetworkType.CONNECTED constraint prevents unnecessary wake |
</threat_model>
